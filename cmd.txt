// Target ECMAScript 2015
tsc --target es2015 hello.ts

// Turn on strict mode 
tsc --strict hello.ts
tsc --noImplicitAny hello.ts
tsc --strictNullChecks hello.ts

tsc functions.ts ; node functions.js ; rm functions.js

sudo ln -s ~/.deno/bin/deno /usr/bin
